  # Start/Stop
  docker-compose up -d                    # Khởi động
  docker-compose stop                     # Dừng
  docker-compose restart                  # Restart tất cả
  
  # Logs
  docker-compose logs -f                  # Xem logs
  docker-compose logs -f student-service  # Logs 1 service
  
  # Build
  docker-compose build                    # Build tất cả
  docker-compose build user-service       # Build 1 service
  
  # Status
  docker-compose ps                       # Trạng thái
  .\health-check.ps1                      # Health check
